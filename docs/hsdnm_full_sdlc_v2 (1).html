Scope enforcement middleware
// packages/rbac/src/scopeQuery.ts import type { Session } from '@hsdnm/auth' export function buildMemberScope(session: Session) { if (!session?.coordinator) throw new Error('Unauthorized') const { role, regionId, stateId, lgaId, wardId } = session.coordinator switch (role) { case 'SUPER_ADMIN': return {} case 'REGIONAL_COORDINATOR': return { regionId } case 'STATE_COORDINATOR': return { stateId } case 'LGA_COORDINATOR': return { lgaId } case 'WARD_COORDINATOR': return { wardId } default: throw new Error('Unauthorized') } } // Usage — scope every query, never trust frontend filters const members = await prisma.user.findMany({ where: { ...buildMemberScope(session), role: 'MEMBER', status: 'ACTIVE' }, include: { state: true, lga: true, ward: true } })

5-step registration flow
1
NIN Entry
11-digit NIN input. Hashed server-side immediately. Duplicate check before any API call.
2
Identity Fetch
NIMC via Dojah/Prembly aggregator. Returns name, DOB, gender. Manual fallback if API down.
3
Profile Review
User confirms pre-filled data. Adds phone (OTP verified), email, profile photo.
4
Geo Selection
Cascade dropdowns: Region → State → LGA → Ward. From internal seeded DB.
5
Membership Card
Unique member ID generated (HSD-SS-BYS-0001). QR code card issued. SMS confirmation.
NIN lookup with resilient fallback
// apps/portal/src/lib/nin.ts async function lookupNIN(nin: string) { const ninHash = sha256(nin + process.env.NIN_HASH_SALT) const existing = await prisma.user.findUnique({ where: { ninHash } }) if (existing) throw new ConflictError('NIN already registered') try { const data = await dojahNINLookup(nin) return { source: 'nimc', ninHash, ...data, requiresManualEntry: false } } catch { return { source: 'manual', ninHash, requiresManualEntry: true } } } // Member ID: HSD-{REGION_CODE}-{STATE_CODE}-{SEQUENCE} function generateMembershipId(regionCode: string, stateCode: string, seq: number) { return `HSD-${regionCode}-${stateCode}-${seq.toString().padStart(4, '0')}` }

! You are handling NIN — one of the most sensitive personal data categories in Nigeria. These controls are non-negotiable before any public launch.
Critical security controls
!NIN never stored raw — SHA-256 with salt, server-side only, original discarded immediately
!RBAC enforced server-side only — never trust frontend filters, scope at Prisma query level
!Supabase RLS policies — Row Level Security mirrors application RBAC as a second defence layer
!Rate-limit NIN endpoint — max 3 attempts per IP per hour, hard lockout after 5
!NDPR compliance — document data purpose, retention policy, member deletion rights
!Audit log every admin action: who acted, on which record, from which IP (manual §10)
!PII fields encrypted at rest (phone, email) using Supabase Vault or pgcrypto
!Profile photos to Supabase Storage only — never user-controlled public URLs
✓Input validation with Zod on every API route
✓HTTPS enforced by Vercel automatically on both domains
✓SUPABASE_SERVICE_ROLE_KEY is server-side only, never exposed to client
✓MemberStatus enum (ACTIVE/SUSPENDED/REMOVED) enforces manual §11 disciplinary process
Supabase RLS example — state coordinator scope
-- State coordinators can SELECT only members in their state CREATE POLICY "state_coord_read_members" ON users FOR SELECT USING ( auth.jwt() ->> 'role' = 'STATE_COORDINATOR' AND state_id = ( SELECT state_id FROM coordinators WHERE user_id = auth.uid() ) ); -- Members can only read their own record CREATE POLICY "member_read_own" ON users FOR SELECT USING (id = auth.uid());
